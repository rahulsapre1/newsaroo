Here are the schemas available in supabase for this newsaroo app

Table 1 - newsaroo_users - This stores the mobile no and topics of interest of the user. 

CREATE TABLE public.newsroom_users (
  id bigint GENERATED BY DEFAULT AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  mobile_number text NOT NULL,
  topics_of_interest jsonb NOT NULL,
  CONSTRAINT newsroom_users_pkey PRIMARY KEY (id, mobile_number)
) TABLESPACE pg_default;
